135  def addPPolicy(self, ppolicyName=None, ppolicyDesc=None):
136  """
137  Add a password policy in LDAP if not available.
138  """
139  if not self.checkPPolicy(ppolicyName):
140 
141  if not ppolicyName:
142  ppolicyName = self.configPPolicy.ppolicydefault
143  ppolicyDesc = "Default password policy"
144  ppolicyDN = self.configPPolicy.ppolicydefaultdn
145  head, path = self.configPPolicy.ppolicydn.split(",", 1)
146  ouName = head.split("=")[1]
147  self.addOu(ouName, path)
148  else:
149  ppolicyDN = "cn=" + ppolicyName + "," + self.configPPolicy.ppolicydn
150 
151  # set common attributes for all ppolicies
152  attrs = {}
153  attrs['objectClass'] = ['pwdPolicy', 'device']
154  attrs['cn'] = ppolicyName
155  if ppolicyDesc:
156  attrs['description'] = ppolicyDesc
157  attrs['pwdattribute'] = self.configPPolicy.ppolicyAttributes['pwdattribute']
158  if 'pwdcheckmodule' in self.configPPolicy.ppolicyAttributes:
159  attrs['objectClass'].append('pwdPolicyChecker')
160  attrs['pwdcheckmodule'] = self.configPPolicy.ppolicyAttributes['pwdcheckmodule']
161 
162  # set default attributes for default password policy
163  if ppolicyName == self.configPPolicy.ppolicydefault:
164  for k in self.configPPolicy.ppolicyAttributes:
165  if type(self.configPPolicy.ppolicyAttributes[k]) == bool:
166  self.configPPolicy.ppolicyAttributes[k] = str(self.configPPolicy.ppolicyAttributes[k]).upper()
167  attrs[k] = str(self.configPPolicy.ppolicyAttributes[k])
168 
169  attributes = modlist.addModlist(attrs)
170  self.l.add_s(ppolicyDN, attributes)
171  logger.info("Password policy registered at: %s" % ppolicyDN)

173  def removePPolicy(self, ppolicyName):

349  def setPPolicyAttribute(self, nameattribute, value):
350  """
351  Set the value of the given LDAP attribute.
352  Del the attribute if value is None
353 
354  @param nameattribute: LDAP attribute name
355  @type nameattribute: str
356 
357  @param value: LDAP attribute value
358  @type value: str
359  """
360  if value is not None:
361  r = AF().log(PLUGIN_NAME, AA.PPOLICY_MOD_USER_ATTR, [(self.dn, AT.USER), (nameattribute, AT.ATTRIBUTE)], value)
362  if type(value) == bool:
363  value = str(value).upper()
364  elif type(value) == int:
365  value = str(value)
366  mode = ldap.MOD_REPLACE
367  logger.debug('Setting %s to %s' % (nameattribute, value))
368  else:
369  r = AF().log(PLUGIN_NAME, AA.PPOLICY_DEL_USER_ATTR, [(self.dn, AT.USER), (nameattribute, AT.ATTRIBUTE)], value)
370  mode = ldap.MOD_DELETE
371  logger.debug('Removing %s' % nameattribute)
372  try:
373  self.l.modify_s(self.dn, [(mode, nameattribute, value)])
374  except ldap.UNDEFINED_TYPE:
375  logger.error("Attribute %s isn't defined on LDAP" % nameattribute)
376  except ldap.INVALID_SYNTAX:
377  logger.error("Invalid Syntax from the attribute value of %s on ldap" % nameattribute)
378  r.commit()
379  # if password reset request, we set sambaPwdLastSet to time()-24hours
380  if nameattribute == 'pwdReset' and value == 'TRUE':
381  userpolicy = self.getPPolicy()
382  if not userpolicy:
383  userpolicy = 'default'
384  pwd_minage = PPolicy().listPPolicy(userpolicy)[0][1]['pwdMinAge'][0]
385  try:
386  from mmc.plugins import samba
387  if samba.isSmbUser(self.userUid):
388  samba.changeSambaAttributes(self.userUid,
389  {'sambaPwdLastSet': str(int(time.time()) - int(pwd_minage))})
390  else:
391  logger.debug('sambaPwdLastSet failed to set \
392  beause %s is not a samba user (pwdReset workaround)' % self.userUid)
393  except ImportError:
394  pass

415  def addPPolicy(self, ppolicyName):
416  """
417  Add the pwdPolicy and pwdPolicySubentry objectClass to the current user,
418  and set the pwdPolicySubentry attribute to the select ppolicy DN
419  """
420  if not self.hasPPolicy():
421  r = AF().log(PLUGIN_NAME, AA.PPOLICY_ADD_USER_PPOLICY, [(self.dn, AT.USER)])
422  # Get current user entry
423  s = self.l.search_s(self.dn, ldap.SCOPE_BASE, attrlist=['+', '*'])
424  c, old = s[0]
425  new = copy.deepcopy(old)
426  if not "pwdPolicy" in new["objectClass"]:
427  new["objectClass"].append("pwdPolicy")
428  new["pwdAttribute"] = "userPassword"
429  new['pwdPolicySubentry'] = PPolicy().getPPolicy(ppolicyName)[0]
430  # Update LDAP
431  modlist = ldap.modlist.modifyModlist(old, new)
432  self.l.modify_s(self.dn, modlist)
433  r.commit()
434  ppolicy_applied.send(sender=self, ppolicy_name=ppolicyName)
435  return True
436  return False

438  def updatePPolicy(self, ppolicyName):
439  """
440  Update the pwdPolicySubentry attribute of the current user
441  """
442  if self.hasPPolicy():
443  if not ppolicyName:
444  return self.removePPolicy()
445  else:
446  # get the ppolicy dn
447  ppolicyDN = PPolicy().getPPolicy(ppolicyName)[0]
448  r = AF().log(PLUGIN_NAME, AA.PPOLICY_MOD_USER_PPOLICY, [(self.dn, AT.USER)])
449  try:
450  self.l.modify_s(self.dn, [(ldap.MOD_REPLACE, 'pwdPolicySubentry', ppolicyDN)])
451  ppolicy_applied.send(sender=self, ppolicy_name=ppolicyName)
452  r.commit()
453  except ldap.UNDEFINED_TYPE:
454  logger.error("Attribute %s isn't defined on ldap" % 'pwdPolicySubentry')
455  except ldap.INVALID_SYNTAX:
456  logger.error("Invalid Syntax from the attribute value of %s on ldap" % 'pwdPolicySubentry')
457  return True
458  else:
459  return self.addPPolicy(ppolicyName)
460 
461  return False

463  def removePPolicy(self):
464  """
465  Remove the pwdPolicy objectClass from the current
466  user, and the pwdPolicySubentry attribute.
467  """
468  r = AF().log(PLUGIN_NAME, AA.PPOLICY_DEL_USER_PPOLICY, [(self.dn, AT.USER)])

479  self.l.modify_s(self.dn, modlist)

481  r.commit()
482  return True

582  def isPasswordExpired(self):
583  """
584  @returns: True if the password is expired
585  @rtype: bool
586  """
587  user = self.getUserEntry(self.userUid, operational=True)
588  ret = False
589  if 'pwdChangedTime' in user:
590  pwdChangedTime = user['pwdChangedTime'][0]
591  if 'pwdMaxAge' in user:
592  pwdMaxAge = user['pwdMaxAge'][0]
593  elif self.getPPolicy():
594  pwdMaxAge = PPolicy().getAttribute('pwdMaxAge', self.getPPolicy())[0]
595  else:
596  pwdMaxAge = PPolicy().getAttribute('pwdMaxAge')[0]
597  if pwdMaxAge == "0" or pwdMaxAge is None:
598  ret = False
599  else:
600  last = calendar.timegm(time.strptime(pwdChangedTime[:-1], '%Y%m%d%H%M%S'))
601  ret = (time.time() - last) > int(pwdMaxAge)
602  return ret
