412  def process(self):
413  """ Processing all the actions """
414  version_to_install = self.ddl_manager.get_version(self.module)
415  if not self.db_exists:
416  self._db_create()
417  version_in_db = 0
418  else:
419  version_in_db = self.db_version
420 
421  if version_in_db == version_to_install:
422  self.log.debug("Database '%s' is up-to-date" % (self.module))
423  return True
424 
425  elif version_in_db < version_to_install:
426  self.log.info("'%s' database updated from version %d to %d" %
427  (self.module, version_in_db, version_to_install))
428  scripts = self._get_scripts_to_install(version_in_db,
429  version_to_install)
430  for script in scripts:
431  if self.script_manager.execute(script) is None:
432  if self.module == 'dyngroup':
433  self.log.warn('Dyngroup known issue: Maybe your SQL engine is MyISAM, you can check with: SHOW TABLE STATUS')
434  self.log.warn('Here is SQL request who will help you to convert from MyISAM to InnoBD engine:')
435  self.log.warn('SELECT CONCAT("ALTER TABLE ",table_schema,".",table_name," ENGINE=InnoDB;") FROM information_schema.tables WHERE table_schema="dyngroup";')
436  return False
437  return True
438  else:
439  self.log.error("Database '%s' version conflict" % self.module)
440  self.log.error("Installed version is %d, but you are trying to install the version %d." %
441  (self.module, version_in_db, version_to_install))
442  return False
