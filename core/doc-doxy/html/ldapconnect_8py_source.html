103  def __init__(self, config):
104  """
105  Set up LDAP connection according to the given configuration.
106 
107  @param config: LDAP connection configuration
108  @type config: LDAPConnectionConfigI
109  """
110  ldap.set_option(ldap.OPT_DEBUG_LEVEL, config.ldapdebuglevel)
111  self.l = ldap.initialize(config.ldapurl)
112  self.l.protocol_version = ldap.VERSION3
113  if config.network_timeout != None:
114  self._ldapSetOption(self.l, ldap.OPT_NETWORK_TIMEOUT, config.network_timeout)
115  if config.start_tls or config.ldapurl.startswith('ldaps://'):
116  # Set SSL/TLS options
117  verifypeer = ldap.OPT_X_TLS_NEVER
118  if config.ldapverifypeer == 'demand':
119  verifypeer = ldap.OPT_X_TLS_DEMAND
120  self._ldapSetOption(self.l, ldap.OPT_X_TLS_REQUIRE_CERT, verifypeer)
121  self._ldapSetOption(self.l, ldap.OPT_X_TLS_CIPHER_SUITE, config.ciphersuites)
122  if config.cacertdir:
123  self._ldapSetOption(self.l, ldap.OPT_X_TLS_CACERTDIR, config.cacertdir)
124  if config.cacert:
125  self._ldapSetOption(self.l, ldap.OPT_X_TLS_CACERTFILE, config.cacert)
126  if config.localcert:
127  self._ldapSetOption(self.l, ldap.OPT_X_TLS_CERTFILE, config.localcert)
128  if config.localkey:
129  self._ldapSetOption(self.l, ldap.OPT_X_TLS_KEYFILE, config.localkey)
130 
131  if config.start_tls:
132  self.l.start_tls_s()
