44  """
45  @param name: the name of the authenticator
46  @param klass: the class name of the authenticator
47  """

51  def validate(self):
52  ret = True
53  tmp = []
54  for name, klass in self.components:
55  mandatory = True
56  try:
57  instance = klass()
58  mandatory = instance.config.mandatory
59  valid = instance.validate()
60  except Exception, e:
61  self.logger.exception(e)
62  valid = False
63  if valid:

65  tmp.append((name, klass))
66  else:
67  self.logger.info("Authenticator %s failed to validate" % name)
68  if mandatory:
69  self.logger.error("Authenticator %s is configured as mandatory, exiting" % name)
70  ret = False

72  self.logger.info("Authenticator %s is not configured as mandatory, so going on" % name)
73  self.components = tmp

86  def authenticate(self, user, password, session):
87  """
88  Loops on the authenticator chains until one successfully authenticates
89  the user.
90 
91  @return: AuthenticationToken object
92  @rtype: AuthenticationToken
93  """
94  token = AuthenticationToken()
95  for name, klass in self.components:
96  instance = klass()
97  self.logger.debug("Try to authenticate user with %s / %s" % (name, str(klass)))
98  if instance.config.authonly:
99  if user.lower() not in instance.config.authonly:
100  self.logger.debug("User %s is not in the authonly list of this authenticator, so we skip it" % user)
101  continue
102  if instance.config.exclude:
103  if user.lower() in instance.config.exclude:
104  self.logger.debug("User %s is in the exclude list of this authenticator, so we skip it" % user)
105  try:
106  token = instance.authenticate(user, password)
107  except Exception, e:
108  self.logger.exception(e)
109  raise AuthenticationError
110  self.logger.debug("Authentication result: " + str(token.authenticated) + str(token.infos))
111  if token.authenticated:
112  # the authentication succeeded
113  break
114  token.session = session
115  return token
