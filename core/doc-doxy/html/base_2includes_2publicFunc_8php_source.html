526 function resizeJpg($source, $maxwidth, $maxheight) {
527  list($width, $height) = getimagesize($source);
528  if (($width > $maxwidth) || ($height > $maxheight)) {
529  if ($width > $height) {
530  $newwidth = $maxwidth;
531  $newheight = $newwidth * $height / $width;
532  } else {
533  $newheight = $maxheight;
534  $newwidth = $newheight * $width / $height;
535  }
536  $image = imagecreatefromjpeg($source);
537  $newimage = imagecreatetruecolor($newwidth, $newheight);
538  imagecopyresized($newimage, $image, 0, 0, 0, 0, $newwidth, $newheight, $width, $height);
539  $ret = tempnam("/notexist", ".jpg");
540  imagejpeg($newimage, $ret);
541  imagedestroy($image);
542  imagedestroy($newimage);
543  } else {
544  /* No resize needed */
545  $ret = $source;
546  }
547  return $ret;
548 }
