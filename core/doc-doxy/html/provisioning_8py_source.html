61  def validate(self):
62  ret = True
63  tmp = []
64  for name, klass in self.components:
65  mandatory = True
66  try:
67  instance = klass()
68  valid = instance.validate()
69  mandatory = instance.config.mandatory
70  except Exception, e:
71  self.logger.exception(e)
72  valid = False

75  tmp.append((name, klass))
76  else:
77  self.logger.info("Provisioner %s failed to validate" % name)
78  if mandatory:
79  self.logger.error("Provisioner %s is configured as mandatory, exiting" % name)
80  ret = False
81  else:

87  def doProvisioning(self, authtoken):
88  """
89  Loops on all the provisioners to perform the provisioning.
90 
91  @param authtoken: AuthenticationToken containing user informations
92  @type authtoken: AuthenticationToken
93 
94  @return; Deferred resulting to authtoken
95  """
96  def _cbError(failure):
97  self.logger.exception(failure.getTraceback())
98  raise ProvisioningError
99 
100  d = None
101  if authtoken.isAuthenticated():
102  login = authtoken.getLogin()
103  for name, klass in self.components:
104  self.logger.debug("Provisioning user with %s / %s" % (name, str(klass)))
105  instance = klass()
106  if login.lower() in instance.config.exclude:
107  self.logger.debug("User %s is in the exclude list of this provisioner, so skipping it" % login)
108  continue
109  if not d:
110  d = defer.maybeDeferred(instance.doProvisioning, authtoken)
111  else:
112  d.addCallback(lambda x: defer.maybeDeferred(instance.doProvisioning, authtoken))
113  if d:
114  ret = d.addCallback(lambda x:authtoken).addErrback(_cbError)
115  else:
116  ret = defer.succeed(authtoken)
117  return ret
