82  def getpassword(self, section, option):
83  """
84  Like get, but interpret the value as a obfuscated password if a
85  password scheme is specified.
86 
87  For example: passwd = {base64}bWFuL2RyaXZhMjAwOA==
88  """
89  value = self.get(section, option)
90  match = re.search('^{(\w+)}(.+)$', value)
91  if match:
92  scheme = match.group(1)
93  obfuscated = match.group(2)
94  ret = obfuscated.decode(scheme)
95  else:
96  ret = value
97  return ret

154 class PluginConfigFactory(object):
155  """
156  For each plugin, we need to have only ONE config instance.
157  This is useful, for example so that the Admin-Configuration plugin
158  can change the config at run time, and every classes using the a PluginConfig will get the new values without restarting.
159  So, every PluginConfig (or a derivated) instance should be created this way
160  """
161  instances = {}
162  @staticmethod
163  def new(cls, name, *args, **kwargs):
164  """
165  If no instance of a class (with this name) has not already
166  been created, create it and keep it in the dict.
167  If one already exist, just return it.
168  """
169  if not name in PluginConfigFactory.instances:
170  PluginConfigFactory.instances[name] = cls(name, *args, **kwargs)
171  return PluginConfigFactory.instances[name]
172 
173  @staticmethod
174  def get(name):
175  """
176  Returns the PluginConfig instance that was
177  created with this name.
178  If it doesn't exist, raise an error because that should never
179  happen.
180  """
181  return PluginConfigFactory.instances.get(name)
