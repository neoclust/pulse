115  def incr(self, key, delta=1):
116  """
117  Add delta to value in the cache. If the key does not exist, raise a
118  ValueError exception.
119  """
120  value = self.get(key)
121  if value is None:
122  raise ValueError("Key '%s' not found" % key)
123  new_value = value + delta
124  self.set(key, new_value)
125  return new_value

130  a ValueError exception.

355  def call(self, *args, **kwargs):
356  # Currently not in progress - start it
357  key = self.hashFunc(*args, **kwargs)
358  if key is None:
359  log.debug("DeferredCache(%s) not hashable: not caching. " % self.op.func_name)
360  return self.op(*args, **kwargs)

386 class RWLock:
387  """
388  Classic implementation of reader-writer lock with preference to writers.
389 
390  Readers can access a resource simultaneously.
391  Writers get an exclusive access.
392 
393  API is self-descriptive:
394  reader_enters()
395  reader_leaves()
396  writer_enters()
397  writer_leaves()
398  """
399  def __init__(self):
400  self.mutex = threading.RLock()
401  self.can_read = threading.Semaphore(0)
402  self.can_write = threading.Semaphore(0)
403  self.active_readers = 0
404  self.active_writers = 0
405  self.waiting_readers = 0
406  self.waiting_writers = 0
407 
408  def reader_enters(self):
409  self.mutex.acquire()
410  try:
411  if self.active_writers == 0 and self.waiting_writers == 0:
412  self.active_readers += 1
413  self.can_read.release()
414  else:
415  self.waiting_readers += 1
416  finally:
417  self.mutex.release()
418  self.can_read.acquire()
419 
420  def reader_leaves(self):
421  self.mutex.acquire()
422  try:
423  self.active_readers -= 1
424  if self.active_readers == 0 and self.waiting_writers != 0:
425  self.active_writers += 1
426  self.waiting_writers -= 1
427  self.can_write.release()
428  finally:
429  self.mutex.release()
430 
431  def writer_enters(self):
432  self.mutex.acquire()
433  try:
434  if self.active_writers == 0 and self.waiting_writers == 0 and self.active_readers == 0:
435  self.active_writers += 1
436  self.can_write.release()
437  else:
438  self.waiting_writers += 1
439  finally:
440  self.mutex.release()
441  self.can_write.acquire()
442 
443  def writer_leaves(self):
444  self.mutex.acquire()
445  try:
446  self.active_writers -= 1
447  if self.waiting_writers != 0:
448  self.active_writers += 1
449  self.waiting_writers -= 1
450  self.can_write.release()
451  elif self.waiting_readers != 0:
452  t = self.waiting_readers
453  self.waiting_readers = 0
454  self.active_readers += t
455  while t > 0:
456  self.can_read.release()
457  t -= 1
458  finally:
459  self.mutex.release()
