98  def _connection_requires_compat(self):
99  # UGLY HACK ALERT. Python 2.7 xmlrpclib caches connection objects in
100  # self._connection (and sets self._connection in __init__). Python
101  # 2.6 and earlier has no such cache. Thus, if self._connection
102  # exists, we're running the newer-style, and if it doesn't then
103  # we're running older-style and thus need compatibility mode.
104  try:
105  self._connection
106  return False
107  except AttributeError:
108  return True

125  def send_cookie_auth(self, connection):
126  """ Include Cookie Authentication data in a header
127  """
128  try:
129  cj = LWPCookieJar()
130  cj.load(COOKIES_FILE, ignore_discard=True, ignore_expires=True)

132  for cookie in cj:
133  connection.putheader('Cookie', '%s=%s' % (cookie.name, cookie.value))
134  return True

154  def request(self, host, handler, request_body, verbose=0):
155  # issue XML-RPC request
156  h = self.make_connection(host)
157  if verbose:
158  h.set_debuglevel(1)
159 
160  self.send_request(h, handler, request_body)

162  self.send_user_agent(h)

164  self.send_content(h, request_body)

166  r = h.getresponse()

168  errmsg = r.reason
169  headers = r.msg
170 
171  # Creati g cookie jar
172  cresponse = CookieResponse(headers)
173  crequest = CookieRequest('https://' + host + '/')
174  if '<methodName>base.ldapAuth</methodName>' in request_body:
175  cj = LWPCookieJar()
176  cj.extract_cookies(cresponse, crequest)
177  if len(cj):
178  cj.save(COOKIES_FILE, ignore_discard=True, ignore_expires=True)
179  os.chmod(COOKIES_FILE, stat.S_IRUSR | stat.S_IWUSR)

181  if errcode != 200:
182  raise xmlrpclib.ProtocolError(
183  host + handler,
184  errcode, errmsg,
185  headers
186  )
187

189
