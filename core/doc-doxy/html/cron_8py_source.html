268 def parseCronLine(line):
269  """
270  Parse a standard cron string (minus the command) and return them as a
271  dictionary. The syntax for this was pulled from the
272  Wikipedia page: http://en.wikipedia.org/wiki/Cron
273  Currently, there is no support for textual days of the week
274  (i.e. Monday,Tuesday).
275  """
276  if not line:
277  raise InvalidCronLine('Empty cron line provided')
278 
279  try:
280  line = re.split('\s+',line.strip())
281  except:
282  raise InvalidCronLine('Cron line must be a string type')
283 
284  if len(line) != 5:
285  raise InvalidCronLine('Improper number of elements encountered: %s' % len(line))
286 
287  schedule = {}
288 
289  schedule['minutes'] = parseCronEntry(line[0],0,59)
290  schedule['hours'] = parseCronEntry(line[1],0,23)
291  schedule['doms'] = parseCronEntry(line[2],1,31)
292  schedule['months'] = parseCronEntry(line[3],1,12)
293  schedule['dows'] = parseCronEntry(line[4],0,6)

297 def parseCronEntry(entry,min,max):
298  """Parse a single cron entry for something like hours or minutes from a cron
299  scheduling line. The given min and max are used to verify that results are
300  in the proper range. The following formats are supports:
301  * => All in the available range
302  */5 => Only those values in the available range that are divisible by five
303  1-5 => The range of 1-5
304  And any combination of the above using commas to separate the entries.
305  """
306 
307  if not entry:
308  raise InvalidCronEntry('Empty cron entry')
309 
310  try:
311  min = int(min)
312  max = int(max)
313  except ValueError:
314  raise ValueError('minimum and maximum should be convertible to integers')
315 
316  if min > max:
317  raise ValueError('minimum must be less than or equal to maximum')
318 
319  if min < 0:
320  raise ValueError('minimum must be non-negative')
321 
322  try:
323  entry = entry.split(',')
324  except AttributeError:
325  raise InvalidCronEntry('Non-string cron-entry')
326 
327  total = set()
328 
329  for e in entry:
330  try:
331  int_val = int(e)
332  except ValueError:
333  pass
334  else:
335  total.add(int_val)
336  continue
337 
338  begin = None
339  end = None
340  step = 1
341 
342  if e == '*':
343  begin = min
344  end = max + 1
345 
346  if begin is None:
347  #If this match works, then it is of the form */int
348  match = _cronStepRe.search(e)
349 
350  if not match is None:
351  begin = min
352  end = max + 1
353  step = int(match.group('step'))
354 
355  if begin is None:
356  match = _cronRangeRe.search(e)
357 
358  if not match is None:
359  begin = int(match.group('begin'))
360  end = int(match.group('end')) + 1
361  step = 1
362 
363  if begin is None:
364  match = _cronRangeStepRe.search(e)
365 
366  if not match is None:
367  begin = int(match.group('begin'))
368  end = int(match.group('end')) + 1
369  step = int(match.group('step'))
370 
371  if (begin is not None and begin < end and step > 0 and
372  begin >= min and end <= max + 1):
373 
374  # need to align the start properly
375  while begin % step != 0 and begin < end:
376  begin += 1
377 
378  if begin == end and begin % step != 0:
379  raise InvalidCronEntry('Invalid range or step specified: %s-%s/%s' % (begin,end,step))
380 
381  total.update(range(begin,end,step))
382  elif not begin is None:
383  raise InvalidCronEntry('Invalid range or step specified: %s-%s/%s' % (begin,end,step))
384 
385  if len(total) == 0:
386  raise InvalidCronEntry('Empty cron entry')
387 
388  total = list(total)
389  total.sort()
390 
391  if total[0] < min or total[len(total)-1] > max:
392  raise InvalidCronEntry('Value, %s-%s, out of allowed range: %s-%s' % (total[0],total[len(total)-1],min,max))
393 
394  return total
