196  # listinfo decorator to handle offsets, limits and output serialization
197  # for XMLRPC, can handle multiple entities queries, multiple columns
198  # and mapped relationships
199  @classmethod
200  def _listinfo(self, func_):
201  @functools.wraps(func_)
202  def __listinfo(self, params, *args, **kw):
203  query = func_(self, params, *args, **kw)
204 
205  # Testing if result is a Query statement
206  if not isinstance(query, Query):
207  logging.getLogger().error('@_listinfo methods must return a Query object, got %s', query.__class__.__name__)
208  return {'count': 0, 'data': [], 'listinfo': 1}
209 
210  # Applying filters on primary entity
211  # Exact filters
212  if 'filters' in params and params['filters']:
213  clauses = [_entity_descriptor(query._mapper_zero(), key) == value
214  for key, value in params['filters'].iteritems()]
215  if clauses:
216  query = query.filter(*clauses)
217  # Like filters
218  if 'like_filters' in params and params['like_filters']:
219  clauses = [_entity_descriptor(query._mapper_zero(), key).like('%' + value + '%')
220  for key, value in params['like_filters'].iteritems()]
221  if clauses:
222  query = query.filter(*clauses)
223 
224  # Calculating count without limit and offset on primary entity id
225  primary_id = _entity_descriptor(query._mapper_zero(), "id")
226  count = query.with_entities(func.count(primary_id))
227  # Scalar doesn't work if multiple entities are selected
228  count = sum([c[0] for c in count.all()])
229 
230  # Applying limit and offset
231  if 'max' in params and 'min' in params:
232  query = query.limit(int(params['max']) - int(params['min'])).offset(int(params['min']))
233 
234  columns = query.column_descriptions
235 
236  data = []
237 
238  # Serializing query output
239  for line in query:
240  if isinstance(line, DBObj):
241  data.append(line.toDict())
242  elif isinstance(line, tuple):
243  # Fetching all tuple items
244  line_ = {}
245  for i in xrange(len(line)):
246  item = line[i]
247  if isinstance(item, DBObj):
248  line_.update(item.toDict())
249  else:
250  if item.__class__.__name__ == 'Decimal':
251  item = int(item)
252  line_.update({columns[i]['name'].encode('ascii', 'ignore') : item})
253  data.append(line_)
254  else:
255  if line.__class__.__name__ == 'Decimal':
256  line = int(line)
257  elif hasattr(line, '_sa_instance_state'):
258  # SA object, try to do a dict conversion
259  line = line.__dict__
260  del line['_sa_instance_state']
261  # Base types
262  data.append(line)
263 
264  # Ensure that session will be closed
265  query.session.close()

336  def toH(self, level=0):
337  ret = {}
338  for i in dir(self):
339  if i in self.i18n:
340  pass
341 
342  if i in self.to_be_exported:
343  ret[i] = getattr(self, i)

345  # we don't want to enter in an infinite loop

347  attr = getattr(self, i)

349  new_attr = []

352  ret[i] = new_attr
