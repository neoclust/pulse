97  def connect(self):
98  """
99  @return: a LDAPobject connected to the LDAP
100  @rtype: LDAPObject
101  """
102  connected = False
103  for ldapurl in self.config.ldapurls:
104  try:
105  self.logger.debug("Connecting to %s" % ldapurl)
106  self.config.ldapurl = ldapurl
107  conn = LDAPConnection(self.config)
108  l = conn.l
109  l.set_option(ldap.OPT_REFERRALS, ldap.OPT_OFF)
110  if self.config.network_timeout:
111  l.set_option(ldap.OPT_NETWORK_TIMEOUT, self.config.network_timeout)
112  if self.config.bindname:
113  l.simple_bind_s(self.config.bindname, self.config.bindpasswd)
114  else:
115  l.simple_bind_s()
116  connected = True
117  except ldap.LDAPError, e:
118  self.logger.info("Can't connect to LDAP server %s %s" % (ldapurl, e))
119  if connected:
120  # Exit loop, because we found a LDAP server to connect to
121  break
122  if not connected:
123  raise Exception("Can't find an external LDAP server to connect to")
124  return l
