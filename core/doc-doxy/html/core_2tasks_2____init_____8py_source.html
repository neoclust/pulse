94  def addTask(self, label, task=None, interval=None, cron_expression=None, delay=0):
95  """
96  Add a task to the task manager.
97 
98  If interval is set a looping call will be run every interval (in sec)
99  If cron expression is set a scheduled call will be run according to the cron expression
100 
101  A deferred is returned to chain errback in case of exception of the task
102  or other callbacks to run then when the task is done or stopped.
103 
104  If the task is already present, it will be restarted.
105 
106  @param label: task name
107  @param task: tuple of (func, [func_args], {func_kwargs})
108  @param interval: interval in seconds for a looping call
109  @param cron_expression: cron expression for a scheduled call
110  @param delay: delay in seconds for a delayed call
111 
112  @return: deferred object to chain errback or callbacks. Callbacks are
113  run when the task is finished. The callbacks take the task as
114  parameter. Errbacks take a twisted Failure instance.
115  """
116  if label in self.tasks:
117  return self.tasks[label]._reschedule()
118 
119  if not task:
120  raise Exception("Can't create any task!")
121 
122  tmp_task = [task[0]]
123  try:
124  tmp_task.append(task[1])
125  except IndexError:
126  tmp_task.append([])
127  try:
128  tmp_task.append(task[2])
129  except IndexError:
130  tmp_task.append({})
131 
132  if interval:
133  self.tasks[label] = LoopingCall(tmp_task[0], *tmp_task[1], **tmp_task[2])
134  logger.debug("Creating interval task %s (interval: %s)" % (label, interval))
135  return self.tasks[label].start(interval)
136 
137  if cron_expression:
138  self.tasks[label] = ScheduledCall(tmp_task[0], *tmp_task[1], **tmp_task[2])
139  logger.debug("Creating cron task %s (cron: %s)" % (label, cron_expression))
140  return self.tasks[label].start(CronSchedule(cron_expression))
141 
142  # If no delay nor interval nor cron_expression
143  # a DelayedCall is run with a 0s delay
144  logger.debug("Creating delayed task %s (delay: %i)" % (label, delay))
145  self.tasks[label] = DelayedCall(tmp_task[0], *tmp_task[1], **tmp_task[2])
146  return self.tasks[label].start(delay)
